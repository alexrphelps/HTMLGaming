// Bresenham's line algorithm to get all grid cells between two points
function getPointsOnLine(x0, y0, x1, y1) {
  const points = [];
  const dx = Math.abs(x1 - x0);
  const dy = Math.abs(y1 - y0);
  const sx = (x0 < x1) ? 1 : -1;
  const sy = (y0 < y1) ? 1 : -1;
  let err = dx - dy;
 
  while (true) {
    points.push({x: x0, y: y0});
    
    if (x0 === x1 && y0 === y1) break;
    
    const e2 = 2 * err;
    if (e2 > -dy) {
      if (x0 === x1) break;
      err -= dy;
      x0 += sx;
    }
    if (e2 < dx) {
      if (y0 === y1) break;
      err += dx;
      y0 += sy;
    }
  }
  
  return points;
}
